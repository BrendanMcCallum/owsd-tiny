cmake_minimum_required(VERSION 2.6)
PROJECT(owsd)
ADD_DEFINITIONS(-Os -Wall --std=gnu99 )

INCLUDE_DIRECTORIES(include)
#FILE(MAKE_DIRECTORY ${CMAKE_CURRENT_SOURCE_DIR}/bin)

SET(SOURCES
	src/main.c
	src/wsubus.c
	src/ws_http.c
	)
